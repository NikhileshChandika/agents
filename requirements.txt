from openai import AzureOpenAI

endpoint = "https://sandbox-sgfj-openai-auseast.openai.azure.com/"

    azure_endpoint=endpoint,
    api_key=subscription_key,

            "role": "user",
            "content": "I am going to Paris, what should I see?",
        }
    ],
    max_tokens=4096,
    temperature=1.0,
    top_p=1.0,
    model=deployment
)
